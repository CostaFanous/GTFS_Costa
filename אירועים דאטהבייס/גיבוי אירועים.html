window.onload = function() { // קבלת התאריך של היום בפורמט yyyy-mm-dd var today = new Date(); var day = ("0" + today.getDate()).slice(-2); var month = ("0" + (today.getMonth() + 1)).slice(-2); var year = today.getFullYear(); var formattedDate = year + "-" + month + "-" + day; // הגדרת תאריך ההתחלה ותאריך הסיום כתאריך של היום document.getElementById('startDate').value = formattedDate; document.getElementById('endDate').value = formattedDate; };
תאריך התחלה תאריך סיום אירוע סוג אירוע מיקום קבצים בפוליגון
